\section{Plots}

\begin{frame}
\frametitle{Mean}
\begin{itemize}
\item
Media
\item
\begin{equation}
\mu = \sum_{i=1}^{n} p_i (x_i)
\end{equation}
$x_i$: event or variable

$p_i$: probability of event $x_i$
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Standard Deviation and Variance}
\begin{itemize}
\item
Variance
\item 
\begin{equation} 
{\sigma}^{2}=\sum_{i=1}^{n} p_i {(x_i - u)}^{2}
\end{equation}
$x_i$: event or variable

$p_i$: probability of event $x_i$

$u$: mean

\item 
Standard Deviation = $\sigma$

\end{itemize}
\end{frame}
